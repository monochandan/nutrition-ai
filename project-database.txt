CREATE TABLE onboarding_questions (
    id SERIAL PRIMARY KEY,
    question_text TEXT NOT NULL,
    question_type VARCHAR(20) NOT NULL, 
	question_subjekt VARCHAR(20) NOT NULL,
    -- 'single_choice', 'multi_choice', 'text', 'number'
    is_required BOOLEAN DEFAULT TRUE,
    order_index INT NOT NULL, -- order of the questions --
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- store choices only for choice based options -- 
CREATE TABLE onboarding_options (
    id SERIAL PRIMARY KEY,
    question_id INT NOT NULL, -- foreign key from onboarding_questions --
    option_text TEXT NOT NULL, -- user see on screen --
    option_value VARCHAR(100), -- backend logic for option_text for my ml uses --
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_question
      FOREIGN KEY (question_id)
      REFERENCES onboarding_questions(id)
      ON DELETE CASCADE
);


CREATE TABLE user_onboarding_answers (
    id SERIAL PRIMARY KEY, -- answer id  --
    user_id INT NOT NULL, -- foreign key from user_account table --
    question_id INT NOT NULL, -- foreign key , question id from onboarding_question table --
    answer_text TEXT, -- answer text typed -- for description
    selected_option_id INT, -- option choose -- for single_choice
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_user_question
      FOREIGN KEY (question_id)
      REFERENCES onboarding_questions(id)
      ON DELETE CASCADE,

    CONSTRAINT fk_selected_option
      FOREIGN KEY (selected_option_id)
      REFERENCES onboarding_options(id)
      ON DELETE SET NULL
);


CREATE TABLE user_onboarding_multi_answers (
    id SERIAL PRIMARY KEY,
    user_id INT NOT NULL, -- foreign key from user_account --
    question_id INT NOT NULL, -- foreign key from onboarding_question table --
    option_id INT NOT NULL,
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_multi_question
      FOREIGN KEY (question_id)
      REFERENCES onboarding_questions(id)
      ON DELETE CASCADE,

    CONSTRAINT fk_multi_option
      FOREIGN KEY (option_id)
      REFERENCES onboarding_options(id)
      ON DELETE CASCADE
);
